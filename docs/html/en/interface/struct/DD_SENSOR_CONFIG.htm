DD_SENSOR_CONFIG

struct of sensor configuration

 struct _dd_sensor_config_{
 unsigned long               iSize;		
 unsigned char	               enable;		
 unsigned char	               bNO;		
 unsigned short	        holdTime;	
 char                        name [DD_MAX_NAME_BUF_LEN];	
 }DD_SENSOR_CONFIG;

Members

iSize

enable
