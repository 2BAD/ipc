DD_TIME_EX

Struct of time configuration information of the device. Compared to DD_TIMEㄛthe month and year value of DD_TIME is different .

 typedef struct _dd_time_ex_
{
    unsigned char	second;		//Seconds after minute (0每59)
    unsigned char	minute;		//Minutes after hour (0每59)
    unsigned char	hour;		//Hours since midnight (0每23)
    unsigned char	wday;		//Day of week (0每6; Sunday = 0)
    unsigned char	mday;		//Day of month (1每31)
    unsigned char	month;		//Month (1每12; January = 1)
    unsigned short	year;		//Year (current year )
    int				nTotalseconds;		//total seconds
    int				nMicrosecond;	//microsecond
}DD_TIME_EX;//Compared to DD_TIMEㄛthe month and year value of DD_TIME is different.
